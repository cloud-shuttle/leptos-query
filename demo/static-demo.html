🚀 What is leptos-query?

leptos-query is a powerful, type-safe data fetching and caching library for Leptos applications, inspired by React Query/TanStack Query. It provides automatic caching, background updates, error handling, and more.

use leptos_query::{use_query, QueryOptions, QueryKey};

let user_query = use_query(
    move || QueryKey::new(["users", &user_id.get().to_string()]),
    move || {
        let id = user_id.get();
        move || async move { fetch_user(id).await }
    },
    QueryOptions::default()
        .with_stale_time(Duration::from_secs(30))
        .with_cache_time(Duration::from_secs(60))
);

✨ Key Features

  • 🔄 Automatic Caching - Queries are automatically cached and shared across components
  • ⚡ Background Updates - Data stays fresh with automatic background refetching
  • 🎯 Query Key-based Invalidation - Precise cache invalidation using query keys
  • 🛡️ Built-in Error Handling - Comprehensive error handling and retry logic
  • 📊 Loading & Stale States - Rich state management for better UX
  • 🔧 Configurable Options - Customize stale times, cache times, retry logic
  • 🚀 Type Safety - Full Rust type safety throughout the API
  • ⚙️ Leptos Integration - Seamless integration with Leptos reactive system

📚 Common Patterns

Dependent Queries

// Fetch user first, then their posts
let user_query = use_query(/* user query */);
let posts_query = use_query(
    move || QueryKey::new(["posts", &user_id.get().to_string()]),
    move || async move { fetch_user_posts(user_id.get()).await },
    QueryOptions::default()
        .enabled(user_query.data.get().is_some()) // Only fetch if user exists
);

Optimistic Updates

let mutation = use_mutation(
    move || async move { update_user(user_id, data).await },
    MutationOptions::default()
        .on_success(move |_| {
            // Invalidate and refetch user data
            query_client.invalidate_queries(&QueryKeyPattern::Exact(
                QueryKey::new(["users", &user_id.to_string()])
            ));
        })
);

Infinite Queries

let infinite_query = use_infinite_query(
    move || QueryKey::new(["posts", "infinite"]),
    move |page_param| async move { 
        fetch_posts_page(page_param).await 
    },
    InfiniteQueryOptions::default()
        .get_next_page_param(|last_page, _pages| last_page.next_cursor)
);

🔧 Configuration Options

QueryOptions::default()
    .with_stale_time(Duration::from_secs(30))      // Data becomes stale after 30s
    .with_cache_time(Duration::from_secs(300))     // Cache expires after 5 minutes
    .with_refetch_interval(Duration::from_secs(60)) // Background refetch every minute
    .with_retry(RetryConfig::default()
        .max_attempts(3)
        .delay(RetryDelay::Exponential {
            initial: Duration::from_secs(1),
            multiplier: 2.0,
            max: Duration::from_secs(30),
        }))
    .keep_previous_data()                          // Show old data while fetching
    .with_suspense()                               // Use Suspense for loading states
